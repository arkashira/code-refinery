"""API for IDE plugin abstraction and registry."""

from __future__ import annotations

import abc
import threading
from typing import Dict, Iterable, List, Optional, Type

from .errors import (
    ConnectionError,
    CommandError,
    UnsupportedIDEError,
)

class IDEPlugin(abc.ABC):
    """Abstract base class for IDE plugins."""

    @abc.abstractmethod
    def connect(self) -> None:
        """Establish a connection to the IDE."""

    @abc.abstractmethod
    def run_command(self, command: str, *args, **kwargs) -> str:
        """Execute a command in the IDE and return its output."""

    @abc.abstractmethod
    def close(self) -> None:
        """Close the connection to the IDE."""

class PluginRegistry:
    """Thread-safe registry for IDE plugins."""

    _lock = threading.RLock()
    _registry: Dict[str, Type[IDEPlugin]] = {}

    @classmethod
    def register(cls, name: str, plugin_cls: Type[IDEPlugin]) -> None:
        """Register a plugin class under a given IDE name."""
        if not issubclass(plugin_cls, IDEPlugin):
            raise TypeError("plugin_cls must inherit from IDEPlugin")
        with cls._lock:
            cls._registry[name.lower()] = plugin_cls

    @classmethod
    def get(cls, name: str) -> Type[IDEPlugin]:
        """Retrieve a plugin class by IDE name."""
        with cls._lock:
            plugin_cls = cls._registry.get(name.lower())
        if plugin_cls is None:
            raise UnsupportedIDEError(f"Unsupported IDE: {name!r}")
        return plugin_cls

    @classmethod
    def list_plugins(cls) -> List[str]:
        """Return a list of registered IDE names."""
        with cls._lock:
            return sorted(cls._registry.keys())

class VSCodePlugin(IDEPlugin):
    """A minimal VSCode plugin implementation."""
