```javascript
// __tests__/vscode.unit.test.ts
import * as vscode from 'vscode';
import { activate, deactivate, runRefinement } from '../src/plugins/vscode';
import { mockBackend } from '../__mocks__/backend';
import { CodeRefineryError } from '../src/plugins/vscode/errors';

jest.mock('../src/plugins/vscode/backend', () => ({
  sendToBackend: jest.fn(),
}));

describe('VS Code plugin unit tests', () => {
  beforeAll(async () => {
    await activate({ subscriptions: [] } as unknown as vscode.ExtensionContext);
  });

  afterAll(() => {
    deactivate();
  });

  test('registers command on activation', () => {
    const commands = (vscode.commands as any)._commands; // internal mock
    expect(commands).toContain('codeRefinery.run');
  });

  test('runRefinement sends active document text to backend', async () => {
    const doc = await vscode.workspace.openTextDocument({ content: 'let a=1;' });
    await vscode.window.showTextDocument(doc);
    (require('../src/plugins/vscode/backend').sendToBackend as jest.Mock)
      .mockResolvedValue({ refined: 'let a = 1;' });

    await runRefinement();

    expect(require('../src/plugins/vscode/backend').sendToBackend)
      .toHaveBeenCalledWith('let a=1;');
  });

  test('displays refined code in new editor tab', async () => {
    const spy = jest.spyOn(vscode.window, 'showTextDocument');
    await runRefinement();
    expect(spy).toHaveBeenCalledWith(
      expect.objectContaining({ getText: expect.any(Function) })
    );
  });

  test('handles missing active editor gracefully', async () => {
    jest.spyOn(vscode.window, 'activeTextEditor').mockReturnValue(undefined);
    const spy = jest.spyOn(vscode.window, 'showErrorMessage');
    await runRefinement();
    expect(spy).toHaveBeenCalledWith(
      expect.stringContaining('No active file')
    );
  });

  test('wraps backend errors in CodeRefineryError', async () => {
    (require('../src/plugins/vscode/backend').sendToBackend as jest.Mock)
      .mockRejectedValue(new Error('Network timeout'));

    await expect(runRefinement()).rejects.toThrow(CodeRefineryError);
  });

  test('uses user configuration with fallback defaults', async () => {
    const cfg = vscode.workspace.getConfiguration('codeRefinery');
    cfg.update('apiEndpoint', 'https://custom.api/', true);
    const endpoint = cfg.get<string>('apiEndpoint');
    expect(endpoint).toBe('https://custom.api/');
    // reset
    await cfg.update('apiEndpoint', undefined, true);
  });
});
```
